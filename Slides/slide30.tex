\begin{frame}{Conclusion}
    \begin{itemize}
        \large
        \setlength\itemsep{0.7em}
        \item Better practices for BBI → successful DFA:
        \begin{itemize}
            \large
            \setlength\itemsep{0.7em}
            \item Impedance matching
            \item Low impedance grounding
        \end{itemize}
        \item Modeling and simulating BBI:
        \begin{itemize}
            \normalsize
            \setlength\itemsep{0.7em}
            \item Local effect on ICs → [100, 200] µm
            \item Thanks to a DC or AC coupling with the probe
            \item Data-dependent faults (bit set and bit reset)
            \item DW substrate / TW positive → dangerous
        \end{itemize}
        \item Substrate thinning and BBI:
        \begin{itemize}
            \normalsize
            \setlength\itemsep{0.7em}
            \item Lowers generator power requirements
            \item Does not change spatial resolution → depends on $(V_{P}, t_{SUB})$
        \end{itemize}
    \end{itemize}
\end{frame}

    \LARGE
    Further improvements:
