
\begin{frame}
    \frametitle{Fault injection attacks}
    \vspace{3mm}
    \begin{Large}Fault injection objectives:\end{Large}
    \vspace{2mm}
    \begin{itemize}
        \item Denial of service (DoS) → Stop circuit operation and the related services
        \item Verification bypass → Modify data on the fly to fake authenticity
        \item Confidential data extraction → Modify data to perform differential fault analysis
    \end{itemize}
    \vspace{3mm}
    \begin{Large}Thanks to a fault injection platform:\end{Large}
    \vspace{2mm}
    \begin{itemize}
        \item Power Glitch Fault Injection (PW-GFI)
        \item Clock Glitch Fault Injection (CK-GFI)
        \item Laser Fault Injection (LFI)
        \item Electromagnetic Fault Injection (EMFI)
        \item \textbf{Body Biasing Fault Injection (BBI)}
    \end{itemize}
\end{frame}
